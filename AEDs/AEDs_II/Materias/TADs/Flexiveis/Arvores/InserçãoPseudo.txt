public void Inserir(int valor) {

	bool ins = false;
	N贸 i = this.raiz;

	if (i == null) {
		i = new N贸(valor);
		ins = true;		
	}

	while (!ins) {
		if (valor < i.valor) {
			if (i.esq == null) {
				i.esq = new N贸(valor);
				ins = true;
			} else {
				i = i.esq;
			}
		} else if (valor > i.valor) {
			if (i.dir == null) {
				i.dir = new N贸(valor);
				ins = true;
			} else {
				i = i.dir;
			}
		} else {
			print("Erro, valor duplicado");
			ins = true;
		}
	}
}
