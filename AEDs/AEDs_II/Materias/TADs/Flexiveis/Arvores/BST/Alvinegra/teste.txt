boolean Inserir(int valor, No pai, No i) {

	if (is4No()) fragmentar();

	No filho == null;

	if (i == null) {
		i = new No(valor);;
	} else if (valor < i.valor) {
		filho = i.esq = Inserir(valor, i.esq);
	} else if () {
		filho = i.dir = Inserir(valor, i.dir);
	} else {
		Erro valor repetido;
	}

	return filho == null ? i : balancear(valor, pai, i, filho);
}

void Inserir(int valor, No avo, No pai, No i) {

	if (is4No()) fragmentar();

	No neto = null;

	if (valor > i.valor) {
		if (i.right == null) neto = i.right = new Node(valor);
		else InsertVoid(valor, i.right);
	} else if (valor < i.valor) {
		if (i.left == null) neto = i.left = new Node(valor);
		else InsertVoid(valor, i.left);
	} else {
		throw new Exception("Erro ao inserir na Árvore: Valor repetido.");
	}

	if (i.cor && neto != null) balancear(avo, pai, i, neto); 
}

void Balancear(No Avo, No pai, No filho, No neto) {

	No raiz;

	if (pai.valor < filho.valor && filho.valor < neto.valor) {
		raiz = balancearSimplesDir();
	} else if (pai.valor < filho.valor && filho.valor > neto.valor) {
		raiz = balancearDuplaDirEsq();
	} else if (pai.valor > filho.valor && filho.valor > neto.valor) {
		raiz = balancearSimplesEsq();
	} else if (pai.valor > filho.valor && filho.valor < neto.valor) {
		raiz = balancearDuplaEsqDir();
	}

	if (avo.valor < raiz.valor) avo.dir = raiz;
	else avo.esq = raiz;
}
